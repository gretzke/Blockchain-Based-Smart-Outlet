%% The Makefile tries to detect dependencies, so this file includes some true

%% commented-out introduction.tex [which doesn't exist]).

%% Also demonstrates the inclusion of revision.tex via the draftfooter package.

\newcount\draft\draft=1 % set to 0 for "publication"

\documentclass{article}
\usepackage{graphicx}

% Generate the glossary and define terms
\usepackage{glossaries}
\makeglossaries
\include{glossary_terms}

\ifnum\draft=1
  % include the revision number in a footer
  \usepackage{draftfooter}
\fi

\begin{document}

\input{abstract.tex} % this comment will be ignored, as will the .tex extension

\begin{figure}[!ht]
  \caption{An SVG picture}
  \centering
    \includegraphics[width=0.5\textwidth]{img/svg.pdf}
\end{figure}

% \input{introduction}
\include{canoe}

% Use glossary items
\section{Time zones}

\section{Write me}
\label{sec:write-me}
Write me.~\cite{www:pdflatex-makefile}

\bibliographystyle{abbrv}
\bibliography{examplebib} % this comment will be ignored

\printglossaries

\end{document}
